CREATE SEQUENCE seq_id_increment
START WITH 1
INCREMENT BY 1;

CREATE TABLE usuario (
  id_usuario INT NOT NULL DEFAULT nextval('seq_id_increment') PRIMARY KEY,
  nombre VARCHAR(20) NOT NULL,
  apellido_1 VARCHAR(20) NOT NULL,
  apellido_2 VARCHAR(20) NOT NULL,
  email VARCHAR(20) NOT NULL,
  usuario VARCHAR(20) NOT NULL,
  password VARCHAR(20) NOT NULL,
  tfn INT(10) NOT NULL
);

CREATE TABLE valoracion (
  id_valoracion INT NOT NULL DEFAULT nextval('seq_id_increment') PRIMARY KEY,
  estrellas INT(5) NOT NULL,
  comentario VARCHAR(40) NOT NULL
);

CREATE TABLE direccion_cliente (
  id_direccion_cliente INT NOT NULL DEFAULT nextval('seq_id_increment') PRIMARY KEY,
  pais VARCHAR(20) NOT NULL,
  provincia VARCHAR(20) NOT NULL,
  ciudad VARCHAR(20) NOT NULL,
  cp VARCHAR(20) NOT NULL,
  direccion VARCHAR(20) NOT NULL
);

CREATE TABLE venta (
  id_venta INT NOT NULL DEFAULT nextval('seq_id_increment') PRIMARY KEY,
  fecha_venta VARCHAR(20) NOT NULL,
  hora_venta VARCHAR(20) NOT NULL
);

CREATE TABLE favoritos (
  id_favoritos INT NOT NULL DEFAULT nextval('seq_id_increment') PRIMARY KEY
);

CREATE TABLE productos (
  id_producto INT NOT NULL DEFAULT nextval('seq_id_increment') PRIMARY KEY,
  marca_producto VARCHAR(20) NOT NULL,
  precio_producto VARCHAR(20) NOT NULL,
  imagen_producto VARCHAR(20) NOT NULL,
  nombre_producto VARCHAR(20) NOT NULL,
  descripcion_producto VARCHAR(20) NOT NULL,
  fecha_subida_producto VARCHAR(20) NOT NULL,
  estado VARCHAR(20) NOT NULL
);

CREATE TABLE producto_categoria (
  id_producto_categoria INT NOT NULL DEFAULT nextval('seq_id_increment') PRIMARY KEY
);

CREATE TABLE categoria (
  id_categoria INT NOT NULL DEFAULT nextval('seq_id_increment') PRIMARY KEY,
  nombre_categoria VARCHAR(20) NOT NULL
);

CREATE TABLE producto_color (
  id_producto_color INT NOT NULL DEFAULT nextval('seq_id_increment') PRIMARY KEY
);

CREATE TABLE color (
  id_color INT NOT NULL DEFAULT nextval('seq_id_increment') PRIMARY KEY,
  nombre_color VARCHAR(20) NOT NULL
);
