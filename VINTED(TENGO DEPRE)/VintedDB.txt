CREATE SEQUENCE seq_idIncrement
   START WITH 1
   INCREMENT BY 1
   NOMAXVALUE
   NOMINVALUE
   NOCACHE
   NOCYCLE;
CREATE TABLE USUARIO(
    IDUSUARIO NUMBER DEFAULT seq_idIncrement.NEXTVAL PRIMARY KEY, 
	NOMBRE VARCHAR2(20),
    APELLIDO_1  VARCHAR2(20),
    APELLIDO_2  VARCHAR2(20),
	EMAIL VARCHAR2(20),
    USUARIO VARCHAR2(20),
    PASSWORD VARCHAR2(20),
    TFN NUMBER(10) 
);
CREATE TABLE VALORACION(
	IDVALORACION NUMBER DEFAULT seq_idIncrement.NEXTVAL PRIMARY KEY,
    ESTRELLASS NUMBER(5),
    COMENTARIO VARCHAR(40)
);
CREATE TABLE DIRECCIONCLIENTE(
	IDDIRECCIONCLIENTE NUMBER DEFAULT seq_idIncrement.NEXTVAL PRIMARY KEY,
    PAIS VARCHAR(20),
    PROVINCIA VARCHAR2(20),
    CIUDAD VARCHAR2(20),
	CP VARCHAR2(20),
    DIRECCION VARCHAR2(20) 
);
CREATE TABLE VENTA(
	IDVENTA NUMBER DEFAULT seq_idIncrement.NEXTVAL PRIMARY KEY,
    FECHA_VENTA VARCHAR(20),
    HORA_VENTA VARCHAR(20)
);
CREATE TABLE FAVORITOS(
	IDFAVORITOS NUMBER DEFAULT seq_idIncrement.NEXTVAL PRIMARY KEY
);
CREATE TABLE PRODUCTOS(
    IDPRODUCTO NUMBER DEFAULT seq_idIncrement.NEXTVAL PRIMARY KEY,
    MARCA_PRODUCTO VARCHAR2(20),
    PRECIO_PRODUCTO VARCHAR2(20),
	IMAGEN_PRODUCTO VARCHAR2(20),
	NOMBRE_PRODUCTO VARCHAR2(20),
	DESCRIPCION_PRODUCTO VARCHAR2(20),
	FECHA_SUBIDA_PRODUCTO VARCHAR2(20),
	ESTADO VARCHAR2(20)
);
CREATE TABLE PRODUCTO_CATEGORIA(
    ID_PRODUCTO_CATEGORIA NUMBER DEFAULT seq_idIncrement.NEXTVAL PRIMARY KEY
);
CREATE TABLE CATEGORIA(
	IDCATEGORIA NUMBER DEFAULT seq_idIncrement.NEXTVAL PRIMARY KEY,
    NOMBRE_CATEGORIA VARCHAR2(20)
);
CREATE TABLE PRODUCTOCOLOR(
    ID_PRODUCTO_COLOR NUMBER DEFAULT seq_idIncrement.NEXTVAL PRIMARY KEY
);
CREATE TABLE COLOR(
    IDCOLOR NUMBER DEFAULT seq_idIncrement.NEXTVAL PRIMARY KEY
    NOMBRE_COLOR VARCHAR2(20)
);
